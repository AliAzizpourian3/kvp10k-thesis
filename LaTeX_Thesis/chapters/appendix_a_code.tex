    """Compute Normalized Edit Distance."""
    # Levenshtein distance
    m, n = len(pred), len(target)
    dp = np.zeros((m+1, n+1), dtype=int)
    
    # ... dynamic programming
    
    edit_distance = dp[m][n]
    ned = edit_distance / max(m, n)
    return ned
